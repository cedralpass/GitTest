git status -- checks status of git

git branch  -- checks branches and current working branch

git branch branch_name -- creates branch

git checkout branch_name -- switches to branch

git add file_name  -- adds branch

git commit -a -m 'adding simple git cheatsheet'  -- commits the current files to the branch

git push origin master / main -- pushes to origin

git push origin branch_name -- pushes new branch_name

git checkout master -- checkout the master / main branch

git merge branch_name -- merges the branch_name into current active branch
